9const BACKEND_URL = import.meta.env.VITE_REACT_APP_BACKEND_URL ?? 'http://localhost:5165';

11// Functie om te controleren of de inlogsessie actief is
12async function checkSession() {
13 try {
14 const response = await fetch(`${BACKEND_URL}/api/Login/CheckSession`, {
15 credentials: 'include',
16 });
17
18 if (!response.ok) {
19 throw new Error('Sessiecontrole mislukt');
20 }
21
22 const data = await response.json();
23 console.log('Sessie Actief: ', data);
24 return data;
25 } catch (error) {
26 console.error('Fout: ', error.message);
27 return null;
28 }
29}

31function PayPage() {
32 const location = useLocation();
33 const navigate = useNavigate();
34 const vehicle = location.state?.vehicle;
35
36 console.log('Voertuig Gegevens:', vehicle);
37
38 const [userDetails, setUserDetails] = useState({
39 email: "",
40 address: "",
41 rentalDates: [null, null],
42 });
43
44 const [errorMessage, setErrorMessage] = useState("");
45 const [totalCost, setTotalCost] = useState(0);
46 const [rentalDays, setRentalDays] = useState(0);
47
48 const handleDateChange = (dates) => {
49 const [start, end] = dates;
50 setUserDetails((prevDetails) => ({ ...prevDetails, rentalDates: [start, end] }));
51
52 if (start && end) {
53 const days = Math.ceil((end - start) / (1000 * 60 * 60 * 24));
54 const pricePerDay = parseFloat(vehicle?.Price.replace(',', '.') || "0");
55
56 if (!isNaN(pricePerDay) && days > 0) {
57 setRentalDays(days);
58 setTotalCost(days * pricePerDay);
59 } else {
60 setRentalDays(0);
61 setTotalCost(0);
62 }
63 } else {
64 setRentalDays(0);
65 setTotalCost(0);
66 }
67 };
68
69 const handlePurchase = async () => {
70 // Controleert of de sessie actief is voordat de huur wordt voortgezet
71 const session = await checkSession();
72
73 if (!session) {
74 setErrorMessage("Log in om door te gaan.");
75 return;
76 }
77
78 // Valideerd of alle vereiste velden zijn ingevuld
79 if (!userDetails.email || !userDetails.address || !userDetails.rentalDates[0] || !userDetails.rentalDates[1]) {
80 setErrorMessage("Vul alle verplichte velden in.");
81 return;
82 }
83
84 // Zorgt ervoor dat FrameNrCar is ingesteld op basis van voertuiggegevens
85 if (!vehicle?.FrameNr) {
86 setErrorMessage("Voertuig heeft geen framenummer. Kies een geldig voertuig.");
87 return;
88 }
89
90 const rentalData = {
91 FrameNrCar: String(vehicle.FrameNr),
92 StartDate: userDetails.rentalDates[0].toISOString(),
93 EndDate: userDetails.rentalDates[1].toISOString(),
94 Price: totalCost,
95 Email: userDetails.email,
96 Address: userDetails.address,
97 };
98
99 console.log("Huurgegevens verzonden:", rentalData);
100
101 try {
102 const response = await fetch(`${BACKEND_URL}/api/Rental/CreateRental`, {
103 method: "POST",
104 headers: {
105 "Content-Type": "application/json",
106 },
107 body: JSON.stringify(rentalData),
108 credentials: "include",
109 });
110
111 if (!response.ok) {
112 const data = await response.json();
113 console.error("Fout bij reactie:", data);
114 setErrorMessage(`Fout: ${data.message}`);
115 return;
116 }
117
118 const data = await response.json();
119 console.log("Huur gemaakt:", data.message);
120 navigate("/confirmationPage", {
121 state: {
122 rental: {
123 vehicleBrand: vehicle.Brand,
124 vehicleType: vehicle.Type,
125 startDate: userDetails.rentalDates[0],
126 endDate: userDetails.rentalDates[1],
127 totalCost: totalCost.toFixed(2),
128 },
129 vehicle,
130 },
131 });
132
133 } catch (error) {
134 console.error("Fout bij het maken van de huur:", error);
135 setErrorMessage("Er is een fout opgetreden bij het verwerken van uw huur. Probeer het opnieuw.");
136 }
137 };
138
139 if (!vehicle) {
140 return (
141 <div className="buy-page">
142 <GeneralHeader />
143 <div className="error-message">
144 <h2>Auto niet gevonden!</h2>
145 <p>Ga terug en selecteer een voertuig.</p>
146 </div>
147 <GeneralFooter />
148 </div>
149 );
150 }
151
152 return (
153 <div className="buy-page">
154 <GeneralHeader />
155 <div className="content">
156 <h1 className="title">Bevestig Rental</h1>
157 <div className="buy-details">
158 <div className="car-info">
159 <h2 className="car-title">{`${vehicle.Brand || "Onbekend"} ${vehicle.Type || "Model"}`}</h2>
160 <p className="car-price">{`Prijs: €${vehicle.Price} per dag`}</p>
161
162 <div className="car-image-container">
163 {vehicle.VehicleBlob ? (
164 <img
165 src={`data:image/jpeg;base64,${vehicle.VehicleBlob}`}
166 alt={`${vehicle.Brand || "Onbekend"} ${vehicle.Type || ""}`}
167 className="car-image"
168 />
169 ) : (
170 <p>Afbeelding niet beschikbaar</p>
171 )}
172 </div>
173 </div>
174 <div className="user-info">
175 <h3 className="user-info-title">Factuuradres en Huurperiode</h3>
176
177 <input
178 type="email"
179 name="email"
180 placeholder="emailadress"
181 value={userDetails.email}
182 onChange={(e) => setUserDetails({ ...userDetails, email: e.target.value })}
183 className="input-field"
184 />
185
186 <input
187 type="text"
188 name="address"
189 placeholder="Vul uw leveradres in"
190 value={userDetails.address}
191 onChange={(e) => setUserDetails({ ...userDetails, address: e.target.value })}
192 className="input-field"
193 />
194
195 <h3>Huurperiode</h3>
196 <DatePicker
197 selected={userDetails.rentalDates[0]}
198 onChange={handleDateChange}
199 startDate={userDetails.rentalDates[0]}
200 endDate={userDetails.rentalDates[1]}
201 selectsRange
202 inline
203 dateFormat="yyyy/MM/dd"
204 placeholderText="Selecteer start- en einddatum"
205 />
206
207 <button
208 className="buy-button"
209 onClick={handlePurchase}
210 >
211 Bevestig Rental
212 </button>
213
214 {errorMessage && <p className="error-message">{errorMessage}</p>}
215 </div>
216 </div>
217
218 {totalCost > 0 && (
219 <div className="total-cost">
220 <h3>Totaal Kosten:</h3>
221 <p>€{totalCost.toFixed(2)}</p>
222 </div>
223 )}
224 </div>
225 <GeneralFooter />
226 </div>
227 );
228}

230export default PayPage;
