9const BACKEND_URL = import.meta.env.VITE_REACT_APP_BACKEND_URL ?? 'http://localhost:5165';

11function ChangeRental() {
12 const [errorMessage, setErrorMessage] = useState("");
13 const [dates, setDates] = useState([null, null]);
14 const [basePrice, setBasePrice] = useState(null);
15 const [totalCost, setTotalCost] = useState(0);
16 const [rentalDays, setRentalDays] = useState(0);
17 const navigate = useNavigate();
18 const location = useLocation();
19 const rental = location.state?.rental;
20
21 const handleDateChange = (dates) => {
22 setDates(dates);
23 const [start, end] = dates;
24
25 if (start && end) {
26 const days = Math.ceil((end - start) / (1000 * 60 * 60 * 24));
27 const pricePerDay = parseFloat(basePrice.replace(',', '.') || "0");
28
29 if (!isNaN(pricePerDay) && days > 0) {
30 setRentalDays(days);
31 setTotalCost(days * pricePerDay);
32 } else {
33 setRentalDays(0);
34 setTotalCost(0);
35 }
36 } else {
37 setRentalDays(0);
38 setTotalCost(0);
39 }
40
41 console.log(rental.id);
42 };
43
44 const handleWijziging = async () => {
45 try {
46 const response = await fetch(`${BACKEND_URL}/api/Rental/ChangeRental`, {
47 method: 'PUT',
48 headers: {
49 'Content-Type': 'application/json',
50 },
51 body: JSON.stringify({
52 Id: rental.id,
53 StartDate: dates[0].toISOString(),
54 EndDate: dates[1].toISOString(),
55 Price: totalCost.toFixed(2),
56 }),
57 });
58
59 if (!response.ok) {
60 const textResponse = await response.text();
61 console.error("Error response:", textResponse);
62 setErrorMessage(`Error: ${textResponse}`);
63 return;
64 }
65
66 setErrorMessage('Rental updated successfully!');
67 navigate("/overviewRental")
68 } catch (error) {
69 console.error("Error when updating rental", error);
70 setErrorMessage("Error when updating rental, please try again later.");
71 }
72 };
73
74 if (!rental) {
75 return <div>Geen rental meegegeven :(</div>
76 }
77
78 useEffect(() => {
79 const fetchBasePrice = async () => {
80 try {
81 const response = await fetch(`${BACKEND_URL}/api/Vehicle/GetVehiclePriceAsync?frameNr=${rental.frameNrCar}`);
82 const data = await response.text();
83 if (response.ok) {
84 setBasePrice(data);

86 setErrorMessage("Failed to fetch base price.");

88 } catch (error) {
89 setErrorMessage("Error fetching base price.");
90 console.error(error);

92 };

94 }, []);

96 console.log(`GIVEN RENTAL IS: ${rental.carName}`)
