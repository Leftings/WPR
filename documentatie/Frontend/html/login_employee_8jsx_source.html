67function Login() {
68 const [userType, SetUserType] = useState('Customer');
69 const [email, setEmail] = useState('');
70 const [password, setPassword] = useState('');
71 const [error, setError] = useState('');
72 const [isMenuOpen, setIsMenuOpen] = useState(false);
73 const navigate = useNavigate();
74
75 const toggleUserType = (event) => {
76 SetUserType(event.target.value);
77 };
78
79 const onSubmit = async (event) => {
80 event.preventDefault();
81
82 if (!email || !password) {
83 if (!email && !password) {
84 setError('Vul een emailadres en wachtwoord in');
85 } else if (!email) {
86 setError('Vul een emailadres in');
87 } else {
88 setError('Vul een wachtwoord in');
89 }
90 return;
91 }
92
93 try
94 {
95 const response = await fetch(`${BACKEND_URL}/api/Login/login`, {
96 method: 'POST',
97 headers: {
98 'Content-Type': 'application/json',
99 },
100 body: JSON.stringify({email, password, userType}),
101 credentials: 'include',
102 });
103
104 if (!response.ok)
105 {
106 setError(`${email} is geen geldig account of het wachtwoord klopt niet`);
107 throw new Error('Login failed');
108 }
109 else
110 {
111 if (userType === 'Customer')
112 {
113 // klant wordt ingelogd terug naar de homepage gestuurd
114 await fetch(`${BACKEND_URL}/api/Login/CheckSession`, { credentials: 'include' });
115 navigate('/');
116 }
117 else if (userType === 'Employee')
118 {
119 // Soort medewerker wordt vastgesteld
120 const officeResponse = await fetch(`${BACKEND_URL}/api/Cookie/GetKindEmployee`, { credentials: 'include' });
121 if (!officeResponse.ok) {
122 throw new Error('Failed to fetch the kind of office');
123 }
124 const office = await officeResponse.json();
125
126 if (office?.message === 'Front')
127 {
128 // Medewerker wordt naar de frontoffice gestuurd
129 await fetch(`${BACKEND_URL}/api/Login/CheckSessionStaff`, { credentials: 'include' });
130 navigate('/FrontOfficeEmployee');
131 }
132 else
133 {
134 // Medewerker wordt naar de backoffice gestuurd
135 await fetch(`${BACKEND_URL}/api/Login/CheckSessionStaff`, { credentials: 'include' });
136 navigate('/BackOfficeEmployee');
137 }
138 }
139 else
140 {
141 // Vehicle Manager wordt naar Vehicle Manager gestuurd
142 await fetch(`${BACKEND_URL}/api/Login/CheckSessionVehicleManager`, { credentials: 'include' });
143 navigate('/VehicleManager');
144 }
145 }
146 }
147 catch (error)
148 {
149 console.error('Error: ', error);
150 }
151 };
152
153 const toggleMenu = () => {
154 setIsMenuOpen(!isMenuOpen);
155 };
156
157 return (
158 <>
159 <link rel="preload" as="image" href={logoHover} />
160 <header className="header">
161 <Link to="/">
162 <div id="left" className="logo-container">
163 <img src={logo} alt="Car And All Logo" className="logo-image"/>
164 <h1 className="logo">Car And All</h1>
165 </div>
166 </Link>
167 <button id = "right" className="hamburger-menu" onClick={toggleMenu}>
168 &#9776; {/* Unicode for hamburger icon */}
169 </button>
170 </header>
171 {isMenuOpen ? (
172 <nav>
173 <ul className="nav-links">
174 <li><Link to="/GeneralSalePage">Zoek Auto's</Link></li>
175 <li><Link to="/about">Over ons</Link></li>
176 <li><Link to="/contact">Contact</Link></li>
177 </ul>
178 </nav>
179 ) : null}
180
181 <main>
182 <div id="inlog">
183 <h1>Login</h1>
184 <div id="input">
185 <label htmlFor="userType">Ik ben een:</label>
186 <br></br>
187 <select id="userType" value={userType} onChange={toggleUserType}>
188 <option value="Customer">Klant</option>
189 <option value="Employee">Medewerker van Car And All</option>
190 <option value="VehicleManager">Wagenpark Beheerder</option>
191 </select>
192 </div>
193 </div>
194
195 <div id="input">
196 <label htmlFor="user">Email</label>
197 <br/>
198 <input type="text" id="user" value={email} onChange={(e) => setEmail(e.target.value)}/>
199 <br/>
200 <label htmlFor="password">Wachtwoord</label>
201 <br/>
202 <input type="password" id="password" value={password} onChange={(e) => setPassword(e.target.value)}/>
203 <br/>
204 <button id="button" type="button" onClick={onSubmit}>Login</button>
205 {userType === "Customer" && (
206 <>
207 <br/>
208 <label htmlFor="noAccount">
209 Nog geen account bij ons? <Link to="/signUp">Meld nu aan!</Link>
210 </label>
211 </>
212 )}
213 {error && <p style={{color: 'red'}}>{error}</p>}

215 </main>
