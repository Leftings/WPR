46 [HttpPost("signUpPersonal")]
47 public async Task<IActionResult> signUpPersonalAsync([FromBody] SignUpRequest signUpRequest)
48 {
49 // Het emailadres wordt gecontrolleerd of het emailadres al bestaat
50 var emailCheck = _userRepository.checkUsageEmailAsync(signUpRequest.Email);
51 bool commit = true;
52
53 using (var connection = _connector.CreateDbConnection())
54 using (var transaction = connection.BeginTransaction())
55 {
56 Console.WriteLine($"Received BirthDate: {signUpRequest.BirthDate}");
57 Console.WriteLine($"{signUpRequest.Email} | {signUpRequest.Password} | {signUpRequest.FirstName} | {signUpRequest.LastName} | {signUpRequest.TelNumber} {signUpRequest.Adres} | {signUpRequest.BirthDate}");
58 try
59 {
60 // Er wordt gekeken of alle gegevens ingevuld zijn
61 bool filledIn = IsFilledIn(signUpRequest);
62 if (!filledIn)
63 {
64 return BadRequest(new { message = "Not all elements are filled in" });
65 }
66
67 // Er wordt gekeken of het opgegeven emailadres valid is
68 if (!EmailChecker.IsValidEmail(signUpRequest.Email))
69 {
70 return BadRequest(new { message = "Invalid email format" });
71 }
72
73 // Er wordt gekeken of er een geldige geboorte datum ingevuld is (niet groter dan 120)
74 if (!BirthdayChecker.IsValidBirthday(signUpRequest.BirthDate))
75 {
76 return BadRequest(new { message = "Invalid birthday format" });
77 }
78
79 // Er wordt gekeken of het ingevoerde telefoonnummer een geldig nederlands nummer is
80 if (!TelChecker.IsValidPhoneNumber(signUpRequest.TelNumber))
81 {
82 return BadRequest(new { message = "Invalid phone number" });
83 }
84
85 // Er wordt gekeken of het wachtwoord bestaat uit minstens 10 karakters, 1 hoofdletter, 1 kleine letter, 1 cijfer en 1 speciaal teken
86 var (isPasswordValid, passwordError) = PasswordChecker.IsValidPassword(signUpRequest.Password);
87 var emailStatus = await emailCheck;
88
89 if (!isPasswordValid)
90 {
91 return BadRequest(new { message = passwordError });
92 }
93
94 // Als het emailadres al bestaat worden de gegevens gerollbacked
95 else if (emailStatus.status)
96 {
97 transaction.Rollback();
98 commit = false;
99
100 return BadRequest( new { message = "Email already exists"} );
101 }
102
103 // Als alle velden zijn ingevuld en de geboorte datum is ingevuld, wordt er een nieuwe customer aangemaakt
104 else if (filledIn && signUpRequest.BirthDate != null)
105 {
106 // Er wordt een nieuwe gebruiker aangemaakt in de tabel UserCustomer
107 var customer = await _userRepository.addCustomerAsync(new object[]
108 {
109 signUpRequest.Adres,
110 signUpRequest.TelNumber,
111 _hash.createHash(signUpRequest.Password),
112 signUpRequest.Email,
113 signUpRequest.FirstName,
114 signUpRequest.LastName
115 });
116
117 // Er wordt aan de nieuwe customer het opgegeven geboortedatum gekoppeld in de tabel UserPersonal

119 {
120 customer.newUserID,
121 signUpRequest.BirthDate
122 });
123
124 // Als het koppelen mislukt wordt de transactie geroldbacked
125 if (!personal.status)
126 {
127 transaction.Rollback();
128 commit = false;
129
130 return BadRequest(new { personal.message });
131 }
132
133 // Er wordt een welkomsemail verstuurd
134 await _emailService.SendWelcomeEmail(signUpRequest.Email);
135
136 return Ok(new { message = "Account created successfully. Please check your email to confirm your account" });
137 }
138 else
139 {
140 return BadRequest(new { message = "Something went wrong (Nothing matched)" });
141 }
142 }
143 catch (Exception ex)
144 {
145 transaction.Rollback();
146 commit = false;
147
148 return StatusCode(500, new
149 {
150 message = ex.Message,
151 details = ex.Message,
152 stackTrace = ex.StackTrace
153 });
154 }
155 finally
156 {
157 if (commit)
158 {
159 transaction.Commit();
160 }
161 }
