34 [HttpGet("GetVehicleNameAsync")]
35 public async Task<IActionResult> GetVehicleName(int frameNr)
36 {
37 try
38 {
39 // Query's om het merk en type van het voertuig op te halen.
40 string query1 = "SELECT Brand FROM Vehicle WHERE FrameNr = @FrameNr";
41 string query2 = "SELECT Type from Vehicle WHERE FrameNr = @FrameNr";
42
43 // Verbinding maken met de database.
44 using (var connection = _connector.CreateDbConnection())
45 {
46 string brand = "";
47 string type = "";
48
49 // Eerste query om het merk van het voertuig op te halen.
50 using (var command = new MySqlCommand(query1, (MySqlConnection)connection))
51 {
52 command.Parameters.AddWithValue("@FrameNr", frameNr);
53
54 using (MySqlDataReader reader1 = await command.ExecuteReaderAsync(CommandBehavior.SingleRow))
55 {
56 // Als het merk gevonden wordt, sla het op in de 'brand' variabele.
57 if (reader1.Read() && !reader1.IsDBNull(0))
58 {
59 brand = reader1.GetString(0);
60 }
61 }
62 }
63
64 // Tweede query om het type van het voertuig op te halen.
65 using (var command2 = new MySqlCommand(query2, (MySqlConnection)connection))
66 {
67 command2.Parameters.AddWithValue("@FrameNr", frameNr);
68
69 using (MySqlDataReader reader2 = await command2.ExecuteReaderAsync(CommandBehavior.SingleRow))
70 {
71 // Als het type gevonden wordt, sla het op in de 'type' variabele en geef een succesvolle response.
72 if (reader2.Read() && !reader2.IsDBNull(0))
73 {
74 type = reader2.GetString(0);
75 return Ok($"{brand} {type}");
76 }
77
78 // Als geen type is gevonden, geef een foutmelding terug.
79 return NotFound($"Car with frameNr {frameNr} not found.");
80 }
81 }
82 }
83 }
84 catch (Exception ex)
85 {
86 // Foutafhandeling, geef een serverfoutmelding terug als er iets misgaat.
87 Console.WriteLine(ex);
88 return StatusCode(500, "An error occurred while fetching the name of the car.");
89 }
90 }

97 [HttpGet("GetVehicleImageAsync")]
98 public async Task<IActionResult> GetVehicleImageAsync(int frameNr)
99 {
100 try
101 {
102 // Query om de afbeelding (VehicleBlob) van het voertuig op te halen.
103 string query = "SELECT VehicleBlob FROM Vehicle WHERE FrameNr = @FrameNr";
104
105 // Verbinding maken met de database.
106 using (var connection = _connector.CreateDbConnection())
107 using (var command = new MySqlCommand(query, (MySqlConnection)connection))
108 {
109 command.Parameters.AddWithValue("@FrameNr", frameNr);
110
111 using (MySqlDataReader reader = await command.ExecuteReaderAsync(CommandBehavior.SingleRow))
112 {
113 // Als de afbeelding is gevonden, converteer het naar Base64 en geef het terug.
114 if (reader.Read() && !reader.IsDBNull(0))
115 {
116 byte[] imageBytes = (byte[])reader["VehicleBlob"];
117 string base64Image = Convert.ToBase64String(imageBytes);
118
119 return Ok(base64Image);
120 }
121 else
122 {
123 // Als geen afbeelding is gevonden, geef een foutmelding terug.
124 return NotFound($"Image with frameNr {frameNr} not found.");
125 }
126 }
127 }
128 }
129 catch (Exception ex)
130 {
131 // Foutafhandeling, geef een serverfoutmelding terug als er iets misgaat.

188 [HttpGet("GetAllVehicles")]
189 public async Task<IActionResult> GetAllVehiclesAsync()
190 {
191 try
192 {
193 string query = @"
194 SELECT FrameNr, YoP, Brand, Type, LicensePlate, Color, Sort, Price, VehicleBlob, Description, Seats
195 FROM Vehicle";
196 var vehicles = new List<object>();
197
198 using (var connection = _connector.CreateDbConnection())
199 using (var command = new MySqlCommand(query, (MySqlConnection)connection))
200 {
201 using (var reader = await command.ExecuteReaderAsync())
202 {
203 while (reader.Read())
204 {
205 vehicles.Add(new
206 {
207 FrameNr = reader.IsDBNull(0) ? 0 : reader.GetInt32(0),
208 YoP = reader.IsDBNull(1) ? 0 : reader.GetInt32(1),
209 Brand = reader.IsDBNull(2) ? null : reader.GetString(2),
210 Type = reader.IsDBNull(3) ? null : reader.GetString(3),
211 LicensePlate = reader.IsDBNull(4) ? null : reader.GetString(4),
212 Color = reader.IsDBNull(5) ? null : reader.GetString(5),
213 Sort = reader.IsDBNull(6) ? null : reader.GetString(6),
214 Price = reader.IsDBNull(7) ? "0.00" : reader.GetDecimal(7).ToString("F2"),
215 Image = reader.IsDBNull(8) ? null : Convert.ToBase64String((byte[])reader["VehicleBlob"]),
216 Description = reader.IsDBNull(9) ? null : reader.GetString(9),
217 Seats = reader.GetInt32(10)
218 });
219 }
220 }
221 }
222
223 return Ok(vehicles);
224 }
225 catch (Exception ex)
226 {
227 Console.WriteLine(ex);
228 return StatusCode(500, "An error occurred while fetching vehicles.");
229 }

263 vehicles.Add(new

265 FrameNr = reader.GetInt32(0),
266 YoP = reader.GetInt32(1),

269 LicensePlate = reader.IsDBNull(4) ? null : reader.GetString(4),
270 Color = reader.IsDBNull(5) ? null : reader.GetString(5),
