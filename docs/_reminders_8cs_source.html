106 public virtual async Task<bool> ReminderContract24Hours()
107 {
108 using (var scope = _serviceProvider.CreateScope())
109 {
110 var customer = scope.ServiceProvider.GetRequiredService<ICustomerDetails>();
111 var contract = scope.ServiceProvider.GetRequiredService<IContractDetails>();
112 var vehicle = scope.ServiceProvider.GetRequiredService<IVehicleDetails>();
113 var contracts = scope.ServiceProvider.GetRequiredService<IContractRepository>();
114
115 if (customer == null || contract == null || vehicle == null || contracts == null)
116 {
117 Console.WriteLine("One or more required services are not available.");
118 return false; // Exit early if any service is not resolved
119 }
120
121 IList<int> ids = await contracts.GetContractsSendEmailAsync();
122
123 if (ids == null || ids.Count == 0)
124 {
125 return false;
126 }
127
128 foreach (int orderId in ids)
129 {
130 Dictionary<string, object> contractDic = new Dictionary<string, object>();
131 Dictionary<string, object> customerDic = new Dictionary<string, object>();
132 Dictionary<string, object> vehicleDic = new Dictionary<string, object>();
133
134 contractDic = await GetContractInfo(orderId, scope);
135 customerDic = await GetCustomerInfo(Convert.ToInt32(contractDic["Customer"]), scope);
136 vehicleDic = await GetVehicleInfo(Convert.ToInt32(contractDic["FrameNrVehicle"]), scope);
137
138 string email = CreateReminderContract(customerDic, contractDic, vehicleDic);
139
140 await SendEmail(email, "Herinnering Huren Voertuig", customerDic["Email"].ToString(), contractDic["OrderId"], scope);
141 }
142
143 return true;
144 }
145 }
